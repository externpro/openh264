cmake_minimum_required(VERSION 3.7.2)
project(openh264)
include(flags OPTIONAL)
if(DEFINED OPENH264_VER)
  set(ver _${OPENH264_VER})
  set(verDir /${PROJECT_NAME}${ver})
  set(prefix ${CMAKE_INSTALL_PREFIX})
  set(VERSION ${OPENH264_VER})
  set(SUFFIX ${ver})
  set(LIBS -lstdc++)
  configure_file(openh264.pc.in openh264.pc @ONLY NEWLINE_STYLE LF)
  install(FILES ${CMAKE_CURRENT_BINARY_DIR}/openh264.pc DESTINATION share/cmake)
endif()
